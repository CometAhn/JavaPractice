



break - 반복문(while, do while)에서 완전하게 벗어나는 문장.
		반복문을 더 이상 실행 안 하겠다.
		int i = 1;
		while(i < 10) {
		i++;
		break; // 조건문 끝내기.
		}
	
continue - 반복문 아래에 있는 문장을 실행하지않고 반복문 처음으로 돌아간다.
		int i = 1;
		while(i < 10) {
		i++;
		continue; // 반복문 처음으로 돌아감.(프린트 실행되지 않음)
		System.out.print("반복문 실험"); // continue 밑이라 에러
		}
